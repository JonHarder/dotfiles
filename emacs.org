* Google hangouts via jabber
#+begin_src emacs-lisp
  (defun jabber-read-config ()
    "Read the encrypted hangouts configuration and load into `jabber-account-list'."
    (interactive)
    (let* ((passfile "~/.emacs.d/jabber.gpg")
            (buf (find-file-noselect passfile))
            (jabber-info (with-current-buffer buf
                           (goto-char (point-min))
                           (read (current-buffer))))
            (email (cdr (assoc 'email jabber-info)))
            (password (cdr (assoc 'password jabber-info))))
       (setq jabber-account-list `((,email
                                    (:password . ,password)
                                    (:network-server . "talk.google.com")
                                    (:connection-type . starttls))))))
  (eval-after-load 'jabber
    '(jabber-read-config))

    "writes the string COMMAND to the pianobar pipe to
  control it remotely"
    (if (stringp command)
        (let ((default-directory "/home/jharder/"))
          (shell-command
           (concat "echo "
                   (when (= (length command) 1) "-n ")
                   "'"
                   command
                   "' > ~/.config/pianobar/ctl")))
      (message "command must be a string")))

    "Searches pdf `FILE' for `REGEX'"
    (interactive (list (read-string (format "Regex [default: %s] " (thing-at-point 'word t)))

  (defun shorten-directory (dir max-length)
    "Show directory name `DIR' taking only the first `MAX-LENGTH' characters."
    (let ((path (reverse (split-string (abbreviate-file-name dir) "/")))
          (output ""))
      (when (and path (equal "" (car path)))
        (setq path (cdr path)))
      (while (and path (< (length output) (- max-length 4)))
        (setq output (concat (car path) "/" output))
        (setq path (cdr path)))
      (when path
        (setq output (concat ".../" output)))
      output))

  (defun shorten-path (path max-len)
    "return a modified version of `path', replacing some components
  with single charactiers starting from the left to try and get the path down to `max-len'"
    (let* ((components (split-string (abbreviate-file-name path) "/"))
           (len (+ (1- (length components))
                   (reduce '+ components :key 'length)))
           (str ""))
      (while (and (> len max-len)
                  (cdr components))
        (setq str (concat str (cond ((= 0 (length (car components))) "/")
                                    ((= 1 (length (car components)))
                                     (concat (car components) "/"))
                                    (t
                                     (if (string= "."
                                                  (string (elt (car components) 0)))
                                         (concat (substring (car components) 0 2)
                                                 "/")
                                       (string (elt (car components) 0) ?/)))))
              len (- len (1- (length (car components))))
              components (cdr components)))
      (concat str (reduce (lambda (a b) (concat a "/" b)) components))))
